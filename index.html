Список команд которые мы использовали :

  1. git config --global core.autocrlf true
  2. git config --global core.safecrlf warn
  3. git config --global core.quotepatch off

  5. git config --global user.email "email"
  6. git init
  7. git branch -m master
  8. git status
  9. git add.
  10. git commit -m "name"
  11. git remote add origin git@github.com:Nikasan1/test_2.git
  12. git branch -M main
  13. git push -u origin main
  14. git pull
